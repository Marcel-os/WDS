226 void GLWidget::initializeGL()
227 {
228  // In this example the widget's corresponding top-level window can change
229  // several times during the widget's lifetime. Whenever this happens, the
230  // QOpenGLWidget's associated context is destroyed and a new one is created.
231  // Therefore we have to be prepared to clean up the resources on the
232  // aboutToBeDestroyed() signal, instead of the destructor. The emission of
233  // the signal will be followed by an invocation of initializeGL() where we
234  // can recreate all resources.
235  connect(context(), &QOpenGLContext::aboutToBeDestroyed, this, &GLWidget::cleanup);
236 
237  initializeOpenGLFunctions();
238  glClearColor(0, 0, 0, m_transparent ? 0 : 1);
239 
240  m_program = new QOpenGLShaderProgram;
241  m_program->addShaderFromSourceCode(QOpenGLShader::Vertex, m_core ? vertexShaderSourceCore : vertexShaderSource);
242  m_program->addShaderFromSourceCode(QOpenGLShader::Fragment, m_core ? fragmentShaderSourceCore : fragmentShaderSource);
243  m_program->bindAttributeLocation("vertex", 0);
244  m_program->bindAttributeLocation("normal", 1);
245  m_program->link();
246 
247  m_program->bind();
248  m_projMatrixLoc = m_program->uniformLocation("projMatrix");
249  m_mvMatrixLoc = m_program->uniformLocation("mvMatrix");
250  m_normalMatrixLoc = m_program->uniformLocation("normalMatrix");
251  m_lightPosLoc = m_program->uniformLocation("lightPos");
252 
253  // Create a vertex array object. In OpenGL ES 2.0 and OpenGL 2.x
254  // implementations this is optional and support may not be present
255  // at all. Nonetheless the below code works in all cases and makes
256  // sure there is a VAO when one is needed.
257  m_vao.create();
258  QOpenGLVertexArrayObject::Binder vaoBinder(&m_vao);
259 
260  // Setup our vertex buffer object.
261  m_RocketVbo.create();
262  m_RocketVbo.bind();
263  m_RocketVbo.allocate(m_Rocket.constData(), m_Rocket.count() * sizeof(GLfloat));
264 
265  // Store the vertex attribute bindings for the program.
266  setupVertexAttribs();
267 
268  // Our camera never changes in this example.
269  m_camera.setToIdentity();
270  m_camera.translate(0, 0, -1.5);
271 
272 
273  // Light position is fixed.
274  m_program->setUniformValue(m_lightPosLoc, QVector3D(0, 0, 10));
275 
276  m_program->release();
277 }

279 void GLWidget::setupVertexAttribs()
280 {
281  m_RocketVbo.bind();
282  QOpenGLFunctions *f = QOpenGLContext::currentContext()->functions();
283  f->glEnableVertexAttribArray(0);
284  f->glEnableVertexAttribArray(1);
285  f->glVertexAttribPointer(0, 3, GL_FLOAT, GL_FALSE, 6 * sizeof(GLfloat), 0);
286  f->glVertexAttribPointer(1, 3, GL_FLOAT, GL_FALSE, 6 * sizeof(GLfloat), reinterpret_cast<void *>(3 * sizeof(GLfloat)));
287  m_RocketVbo.release();
288 }

294 void GLWidget::paintGL()
295 {
296  glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
297  glEnable(GL_DEPTH_TEST);
298  glEnable(GL_CULL_FACE);
299 
300  m_world.setToIdentity();
301  m_world.rotate(180.0f - (m_xRot / 16.0f), 1, 0, 0);
302  m_world.rotate(m_yRot / 16.0f, 0, 1, 0);
303  m_world.rotate(m_zRot / 16.0f, 0, 0, 1);
304 
305  QOpenGLVertexArrayObject::Binder vaoBinder(&m_vao);
306  m_program->bind();
307  m_program->setUniformValue(m_projMatrixLoc, m_proj);
308  m_program->setUniformValue(m_mvMatrixLoc, m_camera * m_world);
309  QMatrix3x3 normalMatrix = m_world.normalMatrix();
310  m_program->setUniformValue(m_normalMatrixLoc, normalMatrix);
311 
312  glDrawArrays(GL_TRIANGLES, 0, m_Rocket.vertexCount());
313 
314  m_program->release();
315 }

340 void GLWidget::mouseMoveEvent(QMouseEvent *event)
341 {
342  int dx = event->x() - m_lastPos.x();
343  int dy = event->y() - m_lastPos.y();
344 
345  if (event->buttons() & Qt::LeftButton) {
346  setXRotation(m_xRot + 8 * dy);
347  setYRotation(m_yRot + 8 * dx);
348  } else if (event->buttons() & Qt::RightButton) {
349  setXRotation(m_xRot + 8 * dy);
350  setZRotation(m_zRot + 8 * dx);
351  }
352  m_lastPos = event->pos();
353 }
